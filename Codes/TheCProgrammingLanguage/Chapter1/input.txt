#include <stdio.h>

// This is a single-line comment
int main() {
    /* This is a
    multi-line
    comment */
    printf("Hello, world!\n"); // Another single-line comment
    /* Nested /* multi-line */ comment */
    printf("/* This is not a comment */\n");
    printf("This /* is not */ a comment\n");
    // Another single-line comment
    /* This is another
    multi-line
    comment */
    /*
    This is a
    big multi-line
    comment
    */
    printf("End of program\n");

    return 0;
}
